			plugins = loader.loadPlugins(new Model(null), null);
		} catch (JAXBException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}

		assertEquals(false, plugins.isEmpty());
		assertArrayEquals(new String[] { "window" }, plugins.get(0).getTags());


	}

}
